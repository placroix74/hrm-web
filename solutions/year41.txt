define label 20 cnt
define label 21 cmp
define label 22 swap
define label 23 input
define label 24 zero
in:
copyfrom zero
copyto cnt
inbox
copyto [cnt]
bumpup cnt
loop:
inbox
jumpz out
copyto input
sort:
copyfrom cnt
copyto swap
copyto cmp
bumpdn cmp
compare:
copyfrom input
sub [cmp]
jumpn insert
swap:
copyfrom [cmp]
copyto [swap]
bumpdn swap
bumpdn cmp
jumpn insert
jump compare
insert:
copyfrom input
copyto [swap]
bumpup cnt
jump loop
out:
bumpdn cnt
jumpn in
copyfrom [cnt]
outbox
jump out
